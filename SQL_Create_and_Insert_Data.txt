CREATE TABLE [Category] (
  [description] Varchar(50),
  PRIMARY KEY ([description])
);

CREATE TABLE [Products] (
  [prodId] INT Identity (1,1),
  [catId] Varchar(50),
  [name] Varchar(50),
  [description] Varchar(200),
  [Size] Varchar(10),
  [unitPrice] decimal(10,2),
  [origin] Varchar(15),
  [image] image,
  PRIMARY KEY ([prodId]),
  CONSTRAINT [FK_Products.catId]
    FOREIGN KEY ([catId])
      REFERENCES [Category]([description])
);

CREATE TABLE [userDetails] (
  [userId] INT,
  [add1] Char(50),
  [add2] Char(50),
  [city] Char(50),
  [zip] Char(10),
  [country] Char(10),
  [phone] Char(10),
  [email] Char(20),
  PRIMARY KEY ([userId])
);

CREATE TABLE [Order] (
  [orderId] INT Identity (1,5),
  [userId] INT,
  [date] DateTime,
  [deliveryCost] decimal(10,2),
  [orderTotal] decimal(10,2),
  [grandTotal] decimal(10,2),
  [stripeId] int,
  PRIMARY KEY ([orderId]),
  CONSTRAINT [FK_Order.userId]
    FOREIGN KEY ([userId])
      REFERENCES [userDetails]([userId])
);

CREATE TABLE [OrderLineItem] (
  [orderId] INT,
  [prodId] INT,
  [qty] INT,
  [lineItemTotal] INT,
  CONSTRAINT [FK_OrderLineItem.orderId]
    FOREIGN KEY ([orderId])
      REFERENCES [Order]([orderId])
);

CREATE TABLE [userStatus] (
  [userId] INT Identity (1,2),
  [firstName] Varchar(35),
  [surName] Varchar(35),
  [gdprConsent] Char(1),
  [status] Varchar(10),
  [type] Char(10)
);
